int fib(int n) {
    int a = 1, b = 1, t;
    if (n < 3) {
        return 1;
    }

    do {
        t = a + b;
        b = a;
        a = t;
        n = n - 1;

        if (t < 233)
        {
            continue;
        }

        #t = 1000;

        if (t > 233)
        {
            break;
        }
    } while(1);

    return t;
}

int main() {
    return fib(35);
}
